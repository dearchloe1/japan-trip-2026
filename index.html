    <script>

        // Firebase Config - REPLACE WITH YOURS

        const firebaseConfig = {

            // YOUR_FIREBASE_CONFIG here e.g.,

            // apiKey: "xxx", authDomain: "xxx.firebaseapp.com", etc.

        };

        firebase.initializeApp(firebaseConfig);

        const db = firebase.database();

        const storage = firebase.storage();

        const auth = firebase.auth();


        // Anon auth for collab

        auth.signInAnonymously().catch(console.error);


        // Online users counter

        const presenceRef = db.ref('/presence');

        const myId = 'user_' + Math.random().toString(36).substr(2, 9);

        presenceRef.child(myId).set(true).onDisconnect(() => presenceRef.child(myId).remove());

        presenceRef.on('value', snap => {

            const count = snap.numChildren();

            document.getElementById('userCount').textContent = `Online: ${count}`;

        });


        // Live Budget

        const budgetRef = db.ref('/budget');

        budgetRef.on('value', snap => {

            const tbody = document.querySelector('#budgetTable tbody');

            tbody.innerHTML = '';

            snap.forEach(child => {

                const data = child.val();

                const row = tbody.insertRow();

                row.innerHTML = `<td contenteditable="true" onblur="updateBudget('${child.key}', 'item', this.textContent)">${data.item}</td>

                                 <td contenteditable="true" onblur="updateBudget('${child.key}', 'paidBy', this.textContent)">${data.paidBy}</td>

                                 <td contenteditable="true" onblur="updateBudget('${child.key}', 'amount', this.textContent)">${data.amount}</td>

                                 <td>${data.receiptUrl ? `<a href="${data.receiptUrl}" target="_blank">View</a>` : ''}</td>

                                 <td contenteditable="true" onblur="updateBudget('${child.key}', 'status', this.textContent)">${data.status || 'Pending'}</td>`;

            });

        });

        document.getElementById('addPaymentForm').onsubmit = async e => {

            e.preventDefault();

            const item = document.getElementById('item').value;

            const paidBy = document.getElementById('paidBy').value;

            const amount = document.getElementById('amount').value;

            const file = document.getElementById('receipt').files[0];

            let receiptUrl = null;

            if (file) {

                const ref = storage.ref(`receipts/${file.name}`);

                await ref.put(file);

                receiptUrl = await ref.getDownloadURL();

            }

            budgetRef.push({item, paidBy, amount, receiptUrl, status: 'Pending'});

            e.target.reset();

        };

        function updateBudget(key, field, value) {

            budgetRef.child(key).update({[field]: value});

        }


        // Live Flights List

        const flightsRef = db.ref('/flights');

        flightsRef.on('value', snap => {

            const list = document.getElementById('flightList');

            list.innerHTML = '';

            snap.forEach(child => {

                const data = child.val();

                const div = document.createElement('div');

                div.className = 'editing';

                div.innerHTML = `<strong>${data.airline}${data.num}</strong>: ${data.status || 'Track'} 

                                 <input placeholder="Status" onblur="updateFlight('${child.key}', 'status', this.value)">

                                 <button onclick="deleteFlight('${child.key}')">Remove</button>`;

                list.appendChild(div);

            });

        });

        document.getElementById('addFlightForm').onsubmit = e => {

            e.preventDefault();

            const num = document.getElementById('flightNum').value;

            const airline = document.getElementById('airline').value;

            flightsRef.push({num, airline});

            e.target.reset();

        };

        function updateFlight(key, field, value) {

            flightsRef.child(key).update({[field]: value});

        }

        function deleteFlight(key) {

            flightsRef.child(key).remove();

        }
