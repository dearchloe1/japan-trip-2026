        async function checkFlight(e) {
            e.preventDefault();
            const num = document.getElementById('flightNum').value;
            const airline = document.getElementById('airline').value;
            // Demo response
            document.getElementById('flightResult').innerHTML = `<p>Flight ${airline}${num}: On time (demo). Use aviationstack API for live.<span class="citation">[web:14]</span></p>`;
        }

        // Budget (localStorage for persistence)
        function addPayment(e) {
            e.preventDefault();
            const item = document.getElementById('item').value;
            const paidBy = document.getElementById('paidBy').value;
            const amount = document.getElementById('amount').value;
            const tbody = document.querySelector('#budgetTable tbody');
            const row = tbody.insertRow();
            row.innerHTML = `
                <td>${item}</td>
                <td>${paidBy}</td>
                <td>${amount}</td>
                <td><input type="file" class="receipt-upload" onchange="uploadReceipt(this)"></td>
                <td><input placeholder="Repaid by"></td>
                <td><select><option>Pending</option><option>Repaid</option></select></td>
            `;
            e.target.reset();
        }
        function uploadReceipt(input) {
            if (input.files[0]) {
                const url = URL.createObjectURL(input.files[0]);
                input.parentNode.innerHTML += `<br><img src="${url}" style="max-width:100px;">`;
            }
        }
